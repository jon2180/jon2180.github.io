    vue封装弹出框组件

    行也成文 8/31/2019 jsvue

    我们一步一步来。

    # 设计需求

    • 允许自定义标题，内容和控制按钮

    # 开始

    根据需求，我们先设计基础结构

    <template>
  <div class="dialog__wrapper"
    v-if="visible"
    :class="{'dialog__wrapper--transparent': !modal}"
    @touchend="touchModal">
    <div class="dialog" :class="{'dialog--center': center}" ref="dialog" :style="popupStyle">
      <div class="dialog__header">
        <span class="dialog__title">{{title}}</span>
        <button class="dialog__headerbtn" style="display: none">关闭</button>
      </div>
      <div class="dialog__body">
        <slot></slot>
      </div>
      <div class="dialog__footer">
        <slot name="footer"></slot>
      </div>
    </div>
  </div>
</template>

<script>
export default {
  name: 'VDialog',
  props: {
    // 是否可见
    visible: {
      type: Boolean,
      default: false,
    },
    // 标题
    title: {
      type: String,
      default: '',
    },
    // 是否居中对齐
    center: {
      type: Boolean,
      default: false,
    },
    width: {
      type: String,
      default: '70%',
    },
    modal: {
      type: Boolean,
      default: true,
    },
    closeOnClickModal: {
      type: Boolean,
      default: true,
    },
    appendToBody: {
      type: Boolean,
      default: false,
    },
  },
  data() {
    return {
      isVisible: this.visible,
      // marginTop: '15vh',
    };
  },
  computed: {
    popupStyle() {
      return {
        // 'margin-top': '15vh',
        // margin: `${this.marginTop} auto`,
        // 'text-align': this.center ? 'center' : 'left',
        width: this.width,
      };
    },
  },
  mounted() {
    if (this.appendToBody) {
      document.body.appendChild(this.$el);
    }
  },
  destroyed() {
    // if appendToBody is true, remove DOM node after destroy
    if (this.appendToBody && this.$el && this.$el.parentNode) {
      this.$el.parentNode.removeChild(this.$el);
    }
  },
  methods: {
    touchModal(e) {
      if (this.visible
        && this.closeOnClickModal
        && !window.document.querySelector('.dialog').contains(e.target)) {
        this.close();
        // this.close();
      }
    },
    close() {
      this.$emit('update:visible', false);
    },
  },
};
</script>
